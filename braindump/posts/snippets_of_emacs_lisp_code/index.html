kill all the buffers that are linked to files with some pattern

(->> (buffer-list)
     (-filter 'buffer-file-name)
     (-map 'buffer-file-name)
     (-filter (-partial 's-contains? "somecontent"))
     (-filter (-partial 's-contains? ".org"))
     (-map 'find-file-noselect)
     (-map 'kill-buffer)
     )
