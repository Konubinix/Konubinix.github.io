void setup() {
  Serial.begin(9600);
  Serial.print("Setting up");
}

void loop() {
  float a0 = analogRead(0);
  float a1 = analogRead(1);
  Serial.print("A0:");
  Serial.print(a0);
  Serial.print(",");
  Serial.print("A1:");
  Serial.print(a1);
  Serial.println();
  delay(10);
}
