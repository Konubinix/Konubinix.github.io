configuring wpa in network interfaces

cat /etc/network/interfaces.d/wlan0

allow-hotplug wlx98ded01db69c
iface wlx98ded01db69c inet dhcp
wpa-conf /etc/wpa_supplicant/wpa_supplicant.conf
iface default inet dhcp

wpa_passphrase myssid mypassphrase

network={
	ssid="myssid"
	#psk="mypassphrase"
	psk=c22c1e6febc7875af85d033bbf15f5ca836633bac8eb16693fd58bff66fcb66c
}

cat /etc/wpa_supplicant/wpa_supplicant.conf

country=fr
ctrl_interface=DIR=/var/run/wpa_supplicant GROUP=netdev
update_config=1
network={
ssid="myssid"
#psk="mypassphrase"
psk=c22c1e6febc7875af85d033bbf15f5ca836633bac8eb16693fd58bff66fcb66c
}

The ctrl_interface stuff was needed on raspbian.
